##
SYS_ACCUEIL_LIENS
=SQL IFNOTEXIST=SYS_ACCUEIL_LIENS=
CREATE TABLE SYS_ACCUEIL_LIENS ( LOGIN VARCHAR2(32) NOT NULL ENABLE, CODE_REGROUP NUMBER(*,3) NOT NULL ENABLE, CODE_CADRE NUMBER(*,3) NOT NULL ENABLE, ORDRE NUMBER(*,3) NOT NULL ENABLE, TYPE VARCHAR2(1) NOT NULL ENABLE, LIEN VARCHAR2(250) NOT NULL ENABLE, LIBELLE VARCHAR2(500) NOT NULL ENABLE, DESCRIPTIF VARCHAR2(250), MESSAGE_ACCUEIL CLOB, ICON VARCHAR2(200) DEFAULT 'fa-external-link', PRIMARY KEY (LOGIN, CODE_REGROUP, CODE_CADRE, ORDRE) )

TESTAUTO_SCENARIO
=SQL IFNOTEXIST=TESTAUTO_SCENARIO=
CREATE TABLE TESTAUTO_SCENARIO ( ID_SCENARIO NUMBER(10,0), LIBELLE VARCHAR2(128), DESCRIPTION VARCHAR2(512), DATE_CREATION DATE, DATE_DERNIERE_MODIF DATE, OWNER VARCHAR2(32), FAIL_ON_ERROR VARCHAR2(1) DEFAULT 'O' NOT NULL ENABLE, SCREENSHOT VARCHAR2(1) DEFAULT 'A', PRIMARY KEY (ID_SCENARIO) )

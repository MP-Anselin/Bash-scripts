SYS_IMPORTSGBD
=SQL IFNOTEXIST=SYS_IMPORTSGBD=
CREATE TABLE SYS_IMPORTSGBD ( DATE_IMPORT VARCHAR2(17) NOT NULL ENABLE, SCHEMA_IMPORT VARCHAR2(50) NOT NULL ENABLE, TYPE_IMPORT VARCHAR2(1) NOT NULL ENABLE, NOM_SCRIPT VARCHAR2(255) NOT NULL ENABLE, TAG_IMPORT VARCHAR2(2048), STATUT VARCHAR2(2) NOT NULL ENABLE, ERREUR VARCHAR2(4000), CHRONO NUMBER DEFAULT 0, PRIMARY KEY (DATE_IMPORT) )

META_PROJET
=SQL IFNOTEXIST=META_PROJET=
CREATE TABLE META_PROJET ( COMPANY_ID VARCHAR2(64) NOT NULL ENABLE, ID_PROJET NUMBER(*,0) NOT NULL ENABLE, LIBELLE_PROJET VARCHAR2(50), DATE_CREATION VARCHAR2(14), USER_CREATION VARCHAR2(32), DESCRIPTION VARCHAR2(4000), TERMINE VARCHAR2(2), NUM_DIT VARCHAR2(50), PRIMARY KEY (COMPANY_ID, ID_PROJET) )

BATCH_QUEUE
=SQL IFNOTEXIST=BATCH_QUEUE=
CREATE TABLE BATCH_QUEUE ( ID_TRT NUMBER(*,3) NOT NULL ENABLE, FAMILLE VARCHAR2(10) NOT NULL ENABLE, REQUESTED_GO VARCHAR2(14) NOT NULL ENABLE, EFFECTIVE_GO VARCHAR2(14), FIRST_GO VARCHAR2(14), NBR_GO NUMBER(*,3), STATE VARCHAR2(8), FORCED_GO VARCHAR2(3), MSGERROR VARCHAR2(255), LOGIN VARCHAR2(32), COMMENTAIRE VARCHAR2(4000), ENDED VARCHAR2(14), PRIMARY KEY (ID_TRT, FAMILLE, REQUESTED_GO) )

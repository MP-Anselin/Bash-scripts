##
NEWS_DETAILS
=SQL IFNOTEXIST=NEWS_DETAILS=
CREATE TABLE "NEWS_DETAILS" ( "CODE" VARCHAR2(32 BYTE), "TYPE" VARCHAR2(12 BYTE), "DESCRIPTION" VARCHAR2(4000 BYTE), "VERSION_RELEASE" VARCHAR2(3 BYTE), "VERSION_MAJEUR" VARCHAR2(3 BYTE), "VERSION_MINEUR" VARCHAR2(3 BYTE), "REPORT" VARCHAR2(4 BYTE), "HOTFIX" VARCHAR2(32 BYTE), "VISIBILITE_CLIENT" VARCHAR2(1 BYTE), CONSTRAINT "PK_NEWS_DETAILS" PRIMARY KEY ("CODE", "VERSION_RELEASE", "VERSION_MAJEUR", "VERSION_MINEUR", "REPORT", "HOTFIX"))

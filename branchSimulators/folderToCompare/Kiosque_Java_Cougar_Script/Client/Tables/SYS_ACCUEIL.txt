##
SYS_ACCUEIL
=SQL IFNOTEXIST=SYS_ACCUEIL=
CREATE TABLE SYS_ACCUEIL ( LOGIN VARCHAR2(32) NOT NULL ENABLE, CODE_REGROUP NUMBER, CODE_CADRE NUMBER, ACTIF VARCHAR2(1), LIBELLE VARCHAR2(255), TYPE_CADRE NUMBER(*,0), ETENDU VARCHAR2(1), COLUMN_FLEX VARCHAR2(1) DEFAULT '1', PRIMARY KEY (LOGIN, CODE_REGROUP, CODE_CADRE) )

##
V_DI_ENTRETIEN_BIL6A_REM
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL6A_REM ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BIL6A_REM","DATE_SITUATION","SALAIRE_BASE","PRCT_AUGMENTATION_IND","PRCT_AUGMENTATION_COL","SAL_MOIS_COMPLET","SAL_AN_COMPLET","SAL_MOIS_PRORAT","SAL_AN_PRORAT","COEF_HORAIRE") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BIL6A_REM",elt."DATE_SITUATION",elt."SALAIRE_BASE",elt."PRCT_AUGMENTATION_IND",elt."PRCT_AUGMENTATION_COL",elt."SAL_MOIS_COMPLET",elt."SAL_AN_COMPLET",elt."SAL_MOIS_PRORAT",elt."SAL_AN_PRORAT",elt."COEF_HORAIRE"
from T_DI_ENTRETIEN_BIL6A_REM elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BIL6A_REM = elt.ID_ENTRETIEN_BIL6A_REM

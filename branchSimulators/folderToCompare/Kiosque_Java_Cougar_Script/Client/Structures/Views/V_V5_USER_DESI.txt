V_V5_USER_DESI
=SQL=
CREATE OR REPLACE VIEW V_V5_USER_DESI AS 
   select  vsu.matricule ,
 vsu.np,
 vsu.matricule_paie,
 'entretien' as identite,
TO_CHAR( vsu.date_sortie ,'dd/mm/yyyy') as date_sortie,
 vsu.nom_prenom_responsable ,
 vsu.nom_prenom_gestionnaire, 
 vsu.nom_prenom_expert, 
 vsu.nom_prenom_directeur_rh,
 vsu.nom_prenom_responsable_rh, 
 vsu.NOM_PRENOM_ACTEUR_INDIRECT from V_V5_USER_DESI_ENTR vsu union select  vsu.matricule ,
 vsu.np,
 vsu.matricule_paie,
 'formation' as identite,
TO_CHAR( vsu.date_sortie, 'dd/mm/yyyy') as date_sortie,
 vsu.nom_prenom_responsable ,
 vsu.nom_prenom_gestionnaire, 
 vsu.nom_prenom_expert, 
 vsu.nom_prenom_directeur_rh,
 vsu.nom_prenom_responsable_rh, 
 vsu.nom_prenom_acteur_indirect from V_V5_USER_DESI_FORM vsu  union select  vsu.matricule ,
 vsu.np,
 vsu.matricule_paie,
 'revue des talents' as identite,
TO_CHAR( vsu.date_sortie , 'dd/mm/yyyy') as date_sortie,
 vsu.nom_prenom_responsable ,
 vsu.nom_prenom_gestionnaire, 
 vsu.nom_prenom_expert, 
 vsu.nom_prenom_directeur_rh,
 vsu.nom_prenom_responsable_rh, 
 vsu.nom_prenom_acteur_indirect from V_V5_USER_DESI_TAL vsu

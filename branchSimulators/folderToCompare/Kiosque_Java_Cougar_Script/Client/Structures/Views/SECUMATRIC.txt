SECUMATRIC
=SQL=
CREATE OR REPLACE VIEW SECUMATRIC AS
SELECT 
	NULL AS NUMPAC,
	u.MATRICULE_PAIE AS PACMAT,
	NULL AS CURRENT_NUMCNT,
	NULL AS DATE_EFF_CONTRAT,
	TO_CHAR(s.MATRICULE) AS MATRIC,
	NULL AS DATE_EFF_PERSO,
	0 AS PRIORITE_SITUATION,
	s.NOM AS NOM,
	s.PRENOM AS PRENOM,
	NULL AS USERMODELE,
	NULL AS CODE_UID,
	NULL AS MATREF,
	s.QUALITE AS TITRE,
	NULL AS CENGES,
	NULL AS NOMJF,
	NULL AS PSEUDO,
	TO_CHAR(s.DATE_NAISSANCE, 'YYYYMMDD') AS DSNAI,
	DECODE(s.SEXE,'M',1,2) AS SEXESS,
	TO_CHAR(s.DATE_NAISSANCE,'RR') AS ANNESS,
	TO_CHAR(s.DATE_NAISSANCE,'MM') AS MOISSS,
	NULL AS DEPTSS,
	NULL AS SECTSS,
	NULL AS ORDRSS,
	NULL AS CLCTSS,
	NULL AS NATCNT,
	NULL AS TYPCNT,
	NULL AS MODCNT,
	NULL AS DSDCN,
	NULL AS DSFCNP,
	NULL AS MOTSUS,
	s.CODE_SOCIETE AS CODSTE,
	s.CODE_ETABLISSEMENT AS CODETB,
	s.CODE_SERVICE AS AFFS02,
	NULL AS CBUD,
	NULL AS NIVECH,
	NULL AS FILIR,
	NULL AS SSFIL,
	s.CODE_METIER AS METIER,
	s.CODE_EMPLOI AS EMPLOI,
	s.LIBELLE_EMPLOI AS QUALPE,
	NULL AS CODCPN,
	s.CODE_CATEGORIE_PRO AS CATPR,
	NULL AS NUGRIL_C,
	NULL AS HOHSAR,
	NULL AS TXACT,
	NULL AS SAFBA,
	NULL AS DSFCN,
	NULL AS MOTCNT,
	s.DATE_SORTIE AS DSSAD,
	NULL AS DSSTC,
	s.DATE_ENTREE_GROUPE AS DSAGP,
	s.DATE_ANCIENNETE_FICTIVE AS DSATO,
	NULL AS NATCOL,
	NULL AS LOTZAD,
	NULL AS DSES1,
	NULL AS DSFSUS,
	NULL AS MODE_DE_PILOTAGE,
	NULL AS ZT061,
	NULL AS TRAV_SECTSS,
	NULL AS ZA700,
	NULL AS PERIODE_RECYCLAGE,
	NULL AS ZA719,
	NULL AS ZAB06,
	NULL AS ZA380,
	NULL AS OLD_AFFS02,
	NULL AS OLD_CODETB,
	NULL AS IDPACMAT,
	NULL AS TITREACC,
	NULL AS ACT_ORG,
	NULL AS BU_ORG,
	NULL AS TRAV_STE,
	NULL AS DSFINC,
	NULL AS MAAFFP,
	NULL AS W_DSFCN
FROM T_V5_KRH_SALARIE s
LEFT JOIN T_V5_KRH_PROFIL_USER u ON s.MATRICULE = u.MATRICULE

##
V_DI_ENTRETIEN_BIL6A_FORM
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL6A_FORM ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BIL6A_FORM","DATE_DEBUT","DATE_FIN","TYPE_ACQUISITION","FORMATION_LIBELLE","STATUT","NBRE_JOURS_REAL","NBRE_HEURES_REAL","OBLIGATOIRE") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BIL6A_FORM",elt."DATE_DEBUT",elt."DATE_FIN",elt."TYPE_ACQUISITION",elt."FORMATION_LIBELLE",elt."STATUT",elt."NBRE_JOURS_REAL",elt."NBRE_HEURES_REAL",elt."OBLIGATOIRE"
from T_DI_ENTRETIEN_BIL6A_FORM elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BIL6A_FORM = elt.ID_ENTRETIEN_BIL6A_FORM

##
V_DI_ENTRETIEN_BILF_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BILF_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BILF_ELT","ORDRE","FORMATION_LIBELLE","FORMATION_OBJECTIF","CODE_STAGE","ORGANISME_LIBELLE","REALISE_NBRE_JOURS","REALISE_NBRE_HEURES","COUT","COMMENTAIRE_COLLAB","COMMENTAIRE_ACTEUR") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BILF_ELT",elt."ORDRE",elt."FORMATION_LIBELLE",elt."FORMATION_OBJECTIF",elt."CODE_STAGE",elt."ORGANISME_LIBELLE",elt."REALISE_NBRE_JOURS",elt."REALISE_NBRE_HEURES",elt."COUT",elt."COMMENTAIRE_COLLAB",elt."COMMENTAIRE_ACTEUR"
from T_DI_ENTRETIEN_BILF_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BILF_ELT = elt.ID_ENTRETIEN_BILF_ELT

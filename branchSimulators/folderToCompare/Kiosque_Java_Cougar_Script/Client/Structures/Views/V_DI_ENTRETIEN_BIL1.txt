##
V_DI_ENTRETIEN_BIL1
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL1 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","BILAN_ACTEUR","BILAN_COLLAB","OBJECTIF","FAITS","AMELIORATIONS") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as BILAN_ACTEUR,
    thm.COMMENTAIRE_COLLAB as BILAN_COLLAB,
    specif_cols.OBJECTIF,
    specif_cols.FAITS,
    specif_cols.AMELIORATIONS
from T_DI_ENTRETIEN_THEME thm
join T_DI_ENTRETIEN_THEME_BIL1 specif_cols on specif_cols.ID_ENTRETIEN_THEME = thm.ID_ENTRETIEN_THEME
where THEME_CODE = 'BIL1 '

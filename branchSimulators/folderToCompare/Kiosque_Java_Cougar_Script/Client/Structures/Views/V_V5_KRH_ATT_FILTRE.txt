##
V_V5_KRH_ATT_FILTRE
=SQL=
CREATE OR REPLACE VIEW V_V5_KRH_ATT_FILTRE ("LIBELLE","ID_FILTRE","ID_POPULATION","ID_STRUCTURE","NUM_RULE_VALIDATION","ID_LETTRE") AS 
SELECT filter.LIBELLE, filter.ID_FILTRE, filter.ID_POPULATION, filter.ID_STRUCTURE,
filter.NUM_RULE_VALIDATION, grp.ID_LETTRE
FROM T_V5_KRH_ATT_FILTRE filter, T_V5_KRH_ATT_GRP_FILTRE grp
WHERE filter.ID_FILTRE = grp.ID_FILTRE

##
V_DI_ENTRETIEN_BIL6A_ENTR
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL6A_ENTR ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BIL6A_ENTR","DATE_ENTRETIEN","TRAME_LIBELLE","STATUT") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BIL6A_ENTR",elt."DATE_ENTRETIEN",elt."TRAME_LIBELLE",elt."STATUT"
from T_DI_ENTRETIEN_BIL6A_ENTR elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BIL6A_ENTR = elt.ID_ENTRETIEN_BIL6A_ENTR

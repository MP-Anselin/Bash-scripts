##
VARSYS_CLIENT
=SQL=
CREATE OR REPLACE VIEW VARSYS_CLIENT AS
SELECT
    S.MODULE AS MODULE_KEY,
    S.CODVAR AS CODVAR_KEY,
    S.MODULE,
    S.CODVAR,
    S.VALVAR,
    S.DESCRIPTION,
    S.TYPVAR,
    DECODE(B.CODVAR, NULL, 'Non', 'Oui') AS BLOQUE
FROM SYS_VARSYS S
LEFT JOIN SYS_VARSYS_BLOQUE B ON S.CODVAR = B.CODVAR

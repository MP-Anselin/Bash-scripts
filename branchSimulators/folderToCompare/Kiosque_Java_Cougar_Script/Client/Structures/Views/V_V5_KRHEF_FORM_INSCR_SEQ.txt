##
V_V5_KRHEF_FORM_INSCR_SEQ
=SQL=
CREATE OR REPLACE VIEW V_V5_KRHEF_FORM_INSCR_SEQ ("ID_INSCRIPTION","ID_INSCRIPTION_SEQ","ID_SESSION_SEQ","ID_PLAN","NUM_SEQ","DATE_SEQ_DEBUT","DATE_SEQ_FIN","SESSION_EFFECTUE","SESSION_SEMI_EFFECTUE","ANNULATION_STAGIAIRE","ANNULATION_ORGANISME","DATE_MAJ") AS 
select
  ID_INSCRIPTION_REF as ID_INSCRIPTION, ID_INSCRIPTION as ID_INSCRIPTION_SEQ, ID_SESSION as ID_SESSION_SEQ, ID_PLAN,
  NUM_SEQ, DATE_SEQ_DEBUT, DATE_SEQ_FIN, SESSION_EFFECTUE, SESSION_SEMI_EFFECTUE, ANNULATION_STAGIAIRE,
  ANNULATION_ORGANISME, DATE_MAJ
from T_V5_KRHEF_FORMATION_INSCR ins
where NUM_SEQ <> 0
  and DATE_INSCRIPTION is not null

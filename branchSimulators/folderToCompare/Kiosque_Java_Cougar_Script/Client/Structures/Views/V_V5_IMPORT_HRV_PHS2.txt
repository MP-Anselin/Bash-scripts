CREATE OR REPLACE VIEW V_V5_IMPORT_HRV_PHS2 ("PERSO","PERSONNALISABLE","ID","APPLI","DESCRIPTION") AS 
select 0 as PERSO, PERSONNALISABLE 
, ID
, APPLI
, DESCRIPTION
 from T_V5_MDL_IMPORT_HRV_PHS2 where ID not in (select ID from T_V5_IMPORT_HRV_PHS2) union all  select 1 as PERSO, 1 as PERSONNALISABLE  
, ID
, APPLI
, DESCRIPTION
 from T_V5_IMPORT_HRV_PHS2 order by ID

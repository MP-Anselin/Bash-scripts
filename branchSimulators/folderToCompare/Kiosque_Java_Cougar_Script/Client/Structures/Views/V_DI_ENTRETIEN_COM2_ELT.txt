##
V_DI_ENTRETIEN_COM2_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_COM2_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_COM2_ELT","ORDRE","LIBELLE","DESCRIPTION","CATEGORIE_LIBELLE","COMMENTAIRE_ACTEUR","COMMENTAIRE_COLLAB") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_COM2_ELT",elt."ORDRE",elt."LIBELLE",elt."DESCRIPTION",elt."CATEGORIE_LIBELLE",elt."COMMENTAIRE_ACTEUR",elt."COMMENTAIRE_COLLAB"
from T_DI_ENTRETIEN_COM2_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_COM2_ELT = elt.ID_ENTRETIEN_COM2_ELT

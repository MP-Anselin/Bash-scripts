##
V_DI_ENTRETIEN_ASP2_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_ASP2_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_ASP2_ELT","ORDRE","MOBILITE_LIBELLE","MOBILITE_DESCRIPTION","ASPIRATION_DESCRIPTION","DELAI_COLLAB_LIBELLE","DELAI_COLLAB_DESCRIPTION","DELAI_ACTEUR_LIBELLE","DELAI_ACTEUR_DESCRIPTION") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_ASP2_ELT",elt."ORDRE",elt."MOBILITE_LIBELLE",elt."MOBILITE_DESCRIPTION",elt."ASPIRATION_DESCRIPTION",elt."DELAI_COLLAB_LIBELLE",elt."DELAI_COLLAB_DESCRIPTION",elt."DELAI_ACTEUR_LIBELLE",elt."DELAI_ACTEUR_DESCRIPTION"
from T_DI_ENTRETIEN_ASP2_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_ASP2_ELT = elt.ID_ENTRETIEN_ASP2_ELT

##
V_V5_KRHEF_CAMPAGNE_INI
=SQL=
CREATE OR REPLACE VIEW V_V5_KRHEF_CAMPAGNE_INI ("ID_CAMPAGNE","ID_CAMPAGNE_INI","DESCRIPTION","DATE_CREATION","DATE_MAJ","DATE_DEBUT_ENTRETIEN","DATE_FIN_ENTRETIEN","ACTIF","COLLABORATIF","STATUTAUTO","STATUTVISU","NIVEAU_VALIDATION","NUM_RULE_VALIDATION","DESCR_COLLABORATEUR","DESCR_MANAGER","DESCR_PDF","ID_BAREME","ID_POPULATION","ID_STRUCTURE","ID_BAREME_SAT","TYPOLOGIE_ABSENCES","ACTEUR","VISU_DETAIL_RESP","ID_TYPOLOGIE","HISTORISE","DATE_ARCHIVAGE","HASH_ARCHIVAGE","IS_ON_DEMAND","VERSION_CAMPAGNE","RELECTURE_OBLIGATOIRE") AS 
select 
  c.ID_CAMPAGNE as ID_CAMPAGNE
, nvl(c.ID_CAMPAGNE_INI, i.ID_CAMPAGNE) as ID_CAMPAGNE_INI
, i.DESCRIPTION
, i.DATE_CREATION
, i.DATE_MAJ
, i.DATE_DEBUT_ENTRETIEN
, i.DATE_FIN_ENTRETIEN
, i.ACTIF
, i.COLLABORATIF
, i.STATUTAUTO
, i.STATUTVISU
, i.NIVEAU_VALIDATION
, i.NUM_RULE_VALIDATION
, i.DESCR_COLLABORATEUR
, i.DESCR_MANAGER
, i.DESCR_PDF
, i.ID_BAREME
, i.ID_POPULATION
, i.ID_STRUCTURE
, i.ID_BAREME_SAT
, i.TYPOLOGIE_ABSENCES
, i.ACTEUR
, i.VISU_DETAIL_RESP
, i.ID_TYPOLOGIE
, i.HISTORISE
, i.DATE_ARCHIVAGE
, i.HASH_ARCHIVAGE
, i.IS_ON_DEMAND
, i.VERSION_CAMPAGNE
, i.RELECTURE_OBLIGATOIRE
 from T_V5_KRHEF_CAMPAGNE c
 join T_V5_KRHEF_CAMPAGNE i on i.ID_CAMPAGNE = nvl(c.ID_CAMPAGNE_INI, c.ID_CAMPAGNE)

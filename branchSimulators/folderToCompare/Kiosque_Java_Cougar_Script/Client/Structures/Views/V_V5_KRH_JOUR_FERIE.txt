##
V_V5_KRH_JOUR_FERIE
=SQL=
CREATE OR REPLACE VIEW V_V5_KRH_JOUR_FERIE ("DATE_JF","CODE","ID_POPULATION","DESCRIPTION","PERSO","ID_JOUR_FERIE") AS 
select DATE_JF, CODE, null as ID_POPULATION, DESCRIPTION, 0 as PERSO, null as ID_JOUR_FERIE
from T_V5_MDL_JOUR_FERIE
where (DATE_JF, CODE) not in (select DATE_JF, CODE
                              from T_V5_KRH_JOUR_FERIE)
union
select DATE_JF, CODE, ID_POPULATION, DESCRIPTION, 1 as PERSO, ID_JOUR_FERIE
from T_V5_KRH_JOUR_FERIE
order by DATE_JF

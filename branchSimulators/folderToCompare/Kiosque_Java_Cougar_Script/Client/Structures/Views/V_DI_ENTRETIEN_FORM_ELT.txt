##
V_DI_ENTRETIEN_FORM_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_FORM_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_FORM_ELT","ORDRE","FORMATION_LIBELLE","DOMAINE_FORMATION","ORGANISME_LIBELLE","OBJECTIF_DEMANDE","PROGRAMME_FORMATION","COUT","CODE_BUDGET","BUDGET_LIBELLE","ORIGINE_DEMANDE","VALIDATION_MANAGER","URGENCE_MANAGER","PREV_JOURS","PREV_HEURES","CPF","DATE_DEMANDE","COMMENTAIRE") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_FORM_ELT",elt."ORDRE",elt."FORMATION_LIBELLE",elt."DOMAINE_FORMATION",elt."ORGANISME_LIBELLE",elt."OBJECTIF_DEMANDE",elt."PROGRAMME_FORMATION",elt."COUT",elt."CODE_BUDGET",elt."BUDGET_LIBELLE",elt."ORIGINE_DEMANDE",elt."VALIDATION_MANAGER",elt."URGENCE_MANAGER",elt."PREV_JOURS",elt."PREV_HEURES",elt."CPF",elt."DATE_DEMANDE",elt."COMMENTAIRE"
from T_DI_ENTRETIEN_FORM_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_FORM_ELT = elt.ID_ENTRETIEN_FORM_ELT

##
V_DI_ENTRETIEN_QUEST_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_QUEST_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_QUEST_ELT","ORDRE","LIBELLE","ELT_TEXTAREA_ACTIF","ELT_TEXTAREA_LIBELLE","ELT_TEXTAREA_ORDRE","ELT_TEXTAREA_REPONSE","ELT_BAREME_ACTIF","ELT_BAREME_ORDRE","ELT_DATE_ACTIF","ELT_DATE_ORDRE","ELT_DATE_REPONSE") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_QUEST_ELT",elt."ORDRE",elt."LIBELLE",elt."ELT_TEXTAREA_ACTIF",elt."ELT_TEXTAREA_LIBELLE",elt."ELT_TEXTAREA_ORDRE",elt."ELT_TEXTAREA_REPONSE",elt."ELT_BAREME_ACTIF",elt."ELT_BAREME_ORDRE",elt."ELT_DATE_ACTIF",elt."ELT_DATE_ORDRE",elt."ELT_DATE_REPONSE"
from T_DI_ENTRETIEN_QUEST_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_QUEST_ELT = elt.ID_ENTRETIEN_QUEST_ELT

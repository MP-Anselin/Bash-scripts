##
V_DI_ENTRETIEN_OBJ1
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_OBJ1 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","OBJECTIF") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as OBJECTIF
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'OBJ1 '

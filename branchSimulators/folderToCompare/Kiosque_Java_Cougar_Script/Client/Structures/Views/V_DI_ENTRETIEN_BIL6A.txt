##
V_DI_ENTRETIEN_BIL6A
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL6A ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","COMMENTAIRE_ENTR","COMMENTAIRE_FORM","COMMENTAIRE_REM","COMMENTAIRE_TYPE_ACQUISITION","COMMENTAIRE_COLLAB","COMMENTAIRE_ACTEUR","DATE_INIT_SNAPSHOT") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    specif_cols.COMMENTAIRE_ENTR,
    specif_cols.COMMENTAIRE_FORM,
    specif_cols.COMMENTAIRE_REM,
    specif_cols.COMMENTAIRE_TYPE_ACQUISITION,
    thm.COMMENTAIRE_COLLAB,
    thm.COMMENTAIRE as COMMENTAIRE_ACTEUR,
    specif_cols.DATE_INIT_SNAPSHOT
from T_DI_ENTRETIEN_THEME thm
join T_DI_ENTRETIEN_THEME_BIL6A specif_cols on specif_cols.ID_ENTRETIEN_THEME = thm.ID_ENTRETIEN_THEME
where THEME_CODE = 'BIL6A'

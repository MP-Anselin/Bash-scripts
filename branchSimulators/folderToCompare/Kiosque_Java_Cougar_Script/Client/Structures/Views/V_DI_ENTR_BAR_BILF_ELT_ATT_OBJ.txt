##
V_DI_ENTR_BAR_BILF_ELT_ATT_OBJ
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTR_BAR_BILF_ELT_ATT_OBJ ("ID_ENTRETIEN_BILF_ELT","ID_BAREME_ITEM","BAREME_LIBELLE","SAISIE_VALEUR","SAISIE_PUCE","SAISIE_ITEM_FR_FR","SAISIE_ITEM_EN_US","SAISIE_ITEM_DE_CH","NB_ITEMS","ORDRE_PUCE","VALEUR_MIN","VALEUR_MAX") AS 
select
    pivot.ID_ENTRETIEN_BILF_ELT,
    bar."ID_BAREME_ITEM",bar."BAREME_LIBELLE",bar."SAISIE_VALEUR",bar."SAISIE_PUCE",bar."SAISIE_ITEM_FR_FR",bar."SAISIE_ITEM_EN_US",bar."SAISIE_ITEM_DE_CH",bar."NB_ITEMS",bar."ORDRE_PUCE",bar."VALEUR_MIN",bar."VALEUR_MAX"
from T_DI_ENTRETIEN_BAREME_SAISIE pivot
join V_DI_BAREME_ITEM_DEF bar on bar.ID_BAREME_ITEM = pivot.ID_BAREME_ITEM
where pivot.TYPE_DE_NOTE = 11

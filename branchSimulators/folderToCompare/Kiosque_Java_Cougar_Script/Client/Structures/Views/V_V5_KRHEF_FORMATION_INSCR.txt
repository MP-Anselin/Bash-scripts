##
V_V5_KRHEF_FORMATION_INSCR
=SQL=
CREATE OR REPLACE VIEW V_V5_KRHEF_FORMATION_INSCR ("ID_INSCRIPTION","ID_SESSION","CODE_STAGE","ID_PLAN","COLLABORATEUR","EXPERT","ID_CATALOGUE","NUM_SEQ","ID_FORMATION_DEMANDE","ID_REF_COLLECTIF","COLLECTIF") AS 
SELECT ins.ID_INSCRIPTION, ins.ID_SESSION, ins.CODE_STAGE, ins.ID_PLAN, ins.COLLABORATEUR,
 ins.EXPERT, ins.ID_CATALOGUE, ins.NUM_SEQ, ins.ID_FORMATION_DEMANDE, dem.ID_REF_COLLECTIF, dem.COLLECTIF
FROM T_V5_KRHEF_FORMATION_INSCR ins, V_V5_KRHEF_FORM_DEMANDE dem
WHERE ins.ID_FORMATION_DEMANDE = dem.ID_FORMATION_DEMANDE

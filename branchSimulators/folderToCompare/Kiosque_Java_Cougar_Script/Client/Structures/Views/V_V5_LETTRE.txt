##
V_V5_LETTRE
=SQL=
CREATE OR REPLACE VIEW V_V5_LETTRE ("PERSO","PERSONNALISABLE","TITRE","ID","APPLI","MODULE","LANGUE","SUPPORT","TYPE","CODE","CONTEXTE","CONTENU_HTML","CONTENU_TXT","ACTIF") AS 
select 0 as PERSO, PERSONNALISABLE, NVL(replace(trim(TITRE), ' '), TYPE||'-'||CODE) AS TITRE 
, ID
, APPLI
, MODULE
, LANGUE
, SUPPORT
, TYPE
, CODE
, CONTEXTE
, CONTENU_HTML
, CONTENU_TXT
, ACTIF
 from T_V5_MDL_LETTRE where ID not in (select ID from T_V5_LETTRE) union all  select 1 as PERSO, 1 as PERSONNALISABLE, NVL(replace(trim(TITRE), ' '), TYPE||'-'||CODE) AS TITRE 
, ID
, APPLI
, MODULE
, LANGUE
, SUPPORT
, TYPE
, CODE
, CONTEXTE
, CONTENU_HTML
, CONTENU_TXT
, ACTIF
 from T_V5_LETTRE order by ID

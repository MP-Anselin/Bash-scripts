##
V_V5_KRH_ATT_SAL_STRUCT
=SQL=
CREATE OR REPLACE VIEW V_V5_KRH_ATT_SAL_STRUCT ("MATRICULE","ID_LETTRE","ID_FILTRE") AS 
SELECT matricule, id_lettre, id_filtre
FROM V_V5_KRH_ATT_FILTRE flt, T_V5_POPULATION_STRUCTURE ps, T_V5_KRH_SALARIE_STRUCTURE struct
WHERE flt.id_structure = ps.id_pop_structure
AND struct.ID_ITEM
IN
(
 SELECT ID_ITEM FROM T_V5_STRUCTURE_ITEM itm
                WHERE ID_STRUCTURE = ps.id_structure
                START WITH
                ID_ITEM = ps.id_item
                CONNECT BY PRIOR itm.ID_ITEM = ID_PARENT_ITEM
)
ORDER BY matricule, id_lettre

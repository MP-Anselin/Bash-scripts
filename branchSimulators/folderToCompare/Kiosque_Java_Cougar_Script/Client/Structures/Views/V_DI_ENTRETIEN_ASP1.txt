##
V_DI_ENTRETIEN_ASP1
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_ASP1 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","ASPIRATIONS") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as ASPIRATIONS
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'ASP1 '

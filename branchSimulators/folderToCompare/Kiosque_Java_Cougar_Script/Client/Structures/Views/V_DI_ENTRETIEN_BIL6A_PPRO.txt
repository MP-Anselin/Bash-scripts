##
V_DI_ENTRETIEN_BIL6A_PPRO
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL6A_PPRO ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BIL6A_PPRO","DATE_DEBUT","DATE_FIN","EMPLOI","CAT_PROF","CONTRAT","COEFFICIENT","CLASSIFICATION") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BIL6A_PPRO",elt."DATE_DEBUT",elt."DATE_FIN",elt."EMPLOI",elt."CAT_PROF",elt."CONTRAT",elt."COEFFICIENT",elt."CLASSIFICATION"
from T_DI_ENTRETIEN_BIL6A_PPRO elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BIL6A_PPRO = elt.ID_ENTRETIEN_BIL6A_PPRO

##
SYS_USR_CONFIG
=SQL=
CREATE OR REPLACE VIEW SYS_USR_CONFIG AS
SELECT
    UPPER(T.VALEUR) AS LOGIN,
    - 1 AS SCHEMA,
    DECODE(UPPER(U.LANGUE), 'FR', 'fr_fr', 'IT', 'it_it',
           U.LANGUE) AS LANGUE,
    TO_CHAR(T.MATRICULE) AS REQUETEMATRIC,
    '|*ALL*|' AS REQ_LISTE_MAT,
    NULL AS REQUETEPW,
    NULL AS DUREEVALIDITEPW,
    NULL AS PW_FIGR,
    NULL AS PARAMSIGNATURE
FROM
    T_V5_KRH_SALARIE_IO_TYPE   T
    INNER JOIN T_V5_KRH_PROFIL_USER       U ON T.MATRICULE = U.MATRICULE
WHERE
    TYPE = 'AOID'

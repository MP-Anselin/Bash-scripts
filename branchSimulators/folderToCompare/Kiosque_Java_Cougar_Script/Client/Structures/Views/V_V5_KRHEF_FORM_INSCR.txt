       ins.SESSION_SEMI_EFFECTUE, ins.ANNULATION_STAGIAIRE, ins.ANNULATION_ORGANISME,
       ins.DATE_ENVOIE_CONVOC, ins.DATE_EDITION_CONVOC, ins.DATE_ENVOIE_ANNUL,
       ins.DATE_EDITION_ANNUL, ins.DATE_EDITION_PECOPCA, ins.DATE_ENVOIE_PAIE, ins.DATE_EDITION_PRESENCE,
       ins.DATE_EVALUATION, ins.EVALUATION_Q1, ins.EVALUATION_Q2, ins.EVALUATION_Q3, ins.EVALUATION_Q4,

       fc.LIBELLE_FORMATION, fc.CODE_STAGE, fc.ID_CATALOGUE,
       org.LIBELLE as LIBELLE_ORGANISME
from
  T_V5_KRHEF_FORMATION_INSCR ins,
  V_V5_KRHEF_SESSION ses,
  T_V5_KRHEF_FORM_CATALOGUE fc,
  T_V5_KRHEF_ORGANISME org
where ins.ID_SESSION = ses.ID_SESSION
  and ses.ID_FORM_CATALOGUE = fc.ID_FORM_CATALOGUE
  and ses.ID_ORGANISME = org.ID_ORGANISME
  and ins.NUM_SEQ = 0
  and ins.DATE_INSCRIPTION is not null

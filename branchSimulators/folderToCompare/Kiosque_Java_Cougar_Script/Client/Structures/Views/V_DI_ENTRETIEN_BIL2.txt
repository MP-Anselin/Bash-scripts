##
V_DI_ENTRETIEN_BIL2
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL2 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","EVAL_GLOBALE_ACTEUR","EVAL_GLOBALE_COLLAB") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as EVAL_GLOBALE_ACTEUR,
    thm.COMMENTAIRE_COLLAB as EVAL_GLOBALE_COLLAB
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'BIL2 '

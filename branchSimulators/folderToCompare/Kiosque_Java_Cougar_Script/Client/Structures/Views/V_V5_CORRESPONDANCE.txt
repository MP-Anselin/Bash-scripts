##
V_V5_CORRESPONDANCE
=SQL=
CREATE OR REPLACE VIEW V_V5_CORRESPONDANCE ("PERSO","PERSONNALISABLE","ID","APPLI","LANGUE","TYPE","GROUPE","NATURE","LIBELLE_REEL","LIBELLE_KIOSQUE","VALEUR_IMPORT","CODE_BASE","ACTIF") AS 
select 0 as PERSO, PERSONNALISABLE 
, ID
, APPLI
, LANGUE
, TYPE
, GROUPE
, NATURE
, LIBELLE_REEL
, LIBELLE_KIOSQUE
, VALEUR_IMPORT
, CODE_BASE
, ACTIF
 from T_V5_MDL_CORRESPONDANCE where ID not in (select ID from T_V5_CORRESPONDANCE) union all  select 1 as PERSO, 1 as PERSONNALISABLE  
, ID
, APPLI
, LANGUE
, TYPE
, GROUPE
, NATURE
, LIBELLE_REEL
, LIBELLE_KIOSQUE
, VALEUR_IMPORT
, CODE_BASE
, ACTIF
 from T_V5_CORRESPONDANCE order by ID

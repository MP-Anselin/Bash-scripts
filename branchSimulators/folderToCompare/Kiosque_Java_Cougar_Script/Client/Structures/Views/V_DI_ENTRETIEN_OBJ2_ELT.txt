##
V_DI_ENTRETIEN_OBJ2_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_OBJ2_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_OBJ2_ELT","ORDRE","LIBELLE","AXE_STRATEG","CATEGORIE","DESCRIPTION","CRITERE_EVAL","INDICATEUR_REALISATION","MOYEN","EMIS_PAR_COLL","DATE_REALISATION","PONDERATION") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_OBJ2_ELT",elt."ORDRE",elt."LIBELLE",elt."AXE_STRATEG",elt."CATEGORIE",elt."DESCRIPTION",elt."CRITERE_EVAL",elt."INDICATEUR_REALISATION",elt."MOYEN",elt."EMIS_PAR_COLL",elt."DATE_REALISATION",elt."PONDERATION"
from T_DI_ENTRETIEN_OBJ2_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_OBJ2_ELT = elt.ID_ENTRETIEN_OBJ2_ELT

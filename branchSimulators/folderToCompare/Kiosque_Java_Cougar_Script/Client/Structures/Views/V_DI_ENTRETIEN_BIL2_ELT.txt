##
V_DI_ENTRETIEN_BIL2_ELT
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_BIL2_ELT ("ID_ENTRETIEN_THEME","ID_ENTRETIEN_BIL2_ELT","ORDRE","OBJ_LIBELLE","OBJ_DESCRIPTION","OBJ_PONDERATION","OBJ_CATEGORIE","OBJ_AXE_STRATEG","OBJ_CRITERE_EVAL","OBJ_INDICATEUR_REALISATION","OBJ_MOYEN","FAIT_MARQUANT") AS 
select
    pivot.ID_ENTRETIEN_THEME,
    elt."ID_ENTRETIEN_BIL2_ELT",elt."ORDRE",elt."OBJ_LIBELLE",elt."OBJ_DESCRIPTION",elt."OBJ_PONDERATION",elt."OBJ_CATEGORIE",elt."OBJ_AXE_STRATEG",elt."OBJ_CRITERE_EVAL",elt."OBJ_INDICATEUR_REALISATION",elt."OBJ_MOYEN",elt."FAIT_MARQUANT"
from T_DI_ENTRETIEN_BIL2_ELT elt
join T_DI_ENTRETIEN_THEME_ELT_SPECF pivot on pivot.ID_ENTRETIEN_BIL2_ELT = elt.ID_ENTRETIEN_BIL2_ELT

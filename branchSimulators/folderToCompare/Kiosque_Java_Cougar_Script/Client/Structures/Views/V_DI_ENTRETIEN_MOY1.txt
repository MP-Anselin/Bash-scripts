##
V_DI_ENTRETIEN_MOY1
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_MOY1 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","COMMENTAIRE_ACTEUR","COMMENTAIRE_COLLAB") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as COMMENTAIRE_ACTEUR,
    thm.COMMENTAIRE_COLLAB
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'MOY1 '

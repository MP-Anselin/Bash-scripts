##
V_DI_ENTRETIEN_MISS2
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_MISS2 ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","TITRE","CATEGORIE","DETAIL_HTML","COMMENTAIRE") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    specif_cols.TITRE,
    specif_cols.CATEGORIE,
    specif_cols.DETAIL_HTML,
    thm.COMMENTAIRE
from T_DI_ENTRETIEN_THEME thm
join T_DI_ENTRETIEN_THEME_MISS2 specif_cols on specif_cols.ID_ENTRETIEN_THEME = thm.ID_ENTRETIEN_THEME
where THEME_CODE = 'MISS2'

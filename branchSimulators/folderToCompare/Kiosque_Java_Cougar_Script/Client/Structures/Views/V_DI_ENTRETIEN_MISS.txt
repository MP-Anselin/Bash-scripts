##
V_DI_ENTRETIEN_MISS
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_MISS ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","MISSION") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.COMMENTAIRE as MISSION
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'MISS '

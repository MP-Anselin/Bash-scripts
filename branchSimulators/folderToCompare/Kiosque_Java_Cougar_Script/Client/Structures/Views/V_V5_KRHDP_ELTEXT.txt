##
V_V5_KRHDP_ELTEXT
=SQL=
CREATE OR REPLACE VIEW V_V5_KRHDP_ELTEXT ("PERSO","ID_ELT_EXT","LANGUE","LECTURE_SEULE","ID_TAB_EXT","MASQUE_PERSO","AFFICHAGE_PERSO","OBLIGATOIRE","MASQUE","AFFICHAGE","IMPORT","EXPORT","CONTROLE_TABLE","CONTROLE_CLIENT","ELT_MULTIPLE","LIBELLE","CODE_FUSION") AS 
select 0 as PERSO 
, ID_ELT_EXT
, LANGUE
, LECTURE_SEULE
, ID_TAB_EXT
, MASQUE_PERSO
, AFFICHAGE_PERSO
, OBLIGATOIRE
, MASQUE
, AFFICHAGE
, IMPORT
, EXPORT
, CONTROLE_TABLE
, CONTROLE_CLIENT
, ELT_MULTIPLE
, LIBELLE
, CODE_FUSION
 from T_V5_MDL_KRHDP_ELTEXT where ID_ELT_EXT not in (select ID_ELT_EXT from T_V5_KRHDP_ELTEXT) union all  select 1 as PERSO 
, ID_ELT_EXT
, LANGUE
, LECTURE_SEULE
, ID_TAB_EXT
, MASQUE_PERSO
, AFFICHAGE_PERSO
, OBLIGATOIRE
, MASQUE
, AFFICHAGE
, IMPORT
, EXPORT
, CONTROLE_TABLE
, CONTROLE_CLIENT
, ELT_MULTIPLE
, LIBELLE
, CODE_FUSION
 from T_V5_KRHDP_ELTEXT order by ID_ELT_EXT

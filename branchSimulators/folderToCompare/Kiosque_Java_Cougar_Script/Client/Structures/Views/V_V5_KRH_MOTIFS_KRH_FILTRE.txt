##
V_V5_KRH_MOTIFS_KRH_FILTRE
=SQL=
CREATE OR REPLACE VIEW V_V5_KRH_MOTIFS_KRH_FILTRE ("LIBELLE","VISIBLE_COLL","VISIBLE_RESP","VISIBLE_GEST","ID_FILTRE","ID_POPULATION","ID_STRUCTURE","NUM_RULE_VALIDATION","NUM_RULE_ANNULATION","NUM_RULE_VALIDATION_AV","NUM_RULE_ANNULATION_AV","ORDRE","MOTIF") AS 
SELECT filter.LIBELLE, filter.VISIBLE_COLL, filter.VISIBLE_RESP, filter.VISIBLE_GEST, filter.ID_FILTRE,
 filter.ID_POPULATION, filter.ID_STRUCTURE, filter.NUM_RULE_VALIDATION, filter.NUM_RULE_ANNULATION,
 filter.NUM_RULE_VALIDATION_AV, filter.NUM_RULE_ANNULATION_AV, filter.ORDRE, grp.MOTIF
FROM T_V5_KRH_MOTIF_KRH_FILTRE filter, T_V5_KRH_MOTIF_KRH_GRP_FILTRE grp
WHERE filter.ID_FILTRE = grp.ID_FILTRE

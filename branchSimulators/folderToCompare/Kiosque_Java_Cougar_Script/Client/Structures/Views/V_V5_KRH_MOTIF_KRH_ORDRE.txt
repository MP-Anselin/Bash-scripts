##
V_V5_KRH_MOTIF_KRH_ORDRE
=SQL=
CREATE OR REPLACE VIEW V_V5_KRH_MOTIF_KRH_ORDRE ("MOTIF","ORDRE") AS 
select distinct MOTIF
, ORDRE
 from T_V5_MDL_MOTIF_KRH where MOTIF not in (select MOTIF from T_V5_KRH_MOTIF_KRH_ORDRE) union all select MOTIF
, ORDRE
 from T_V5_KRH_MOTIF_KRH_ORDRE order by ORDRE

##
V_DI_ENTRETIEN_FORM
=SQL=
CREATE OR REPLACE VIEW V_DI_ENTRETIEN_FORM ("ID_ENTRETIEN_THEME","ID_ENTRETIEN","THEME_LIBELLE","ORDRE","AUCUN_SOUHAIT") AS 
select
    thm.ID_ENTRETIEN_THEME,
    thm.ID_ENTRETIEN,
    thm.THEME_LIBELLE,
    thm.ORDRE,
    thm.AUCUN_SOUHAIT
from T_DI_ENTRETIEN_THEME thm
where THEME_CODE = 'FORM '

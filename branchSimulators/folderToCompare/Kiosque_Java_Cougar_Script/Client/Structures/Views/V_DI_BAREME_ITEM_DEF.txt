##
V_DI_BAREME_ITEM_DEF
=SQL=
CREATE OR REPLACE VIEW V_DI_BAREME_ITEM_DEF ("ID_BAREME_ITEM","BAREME_LIBELLE","SAISIE_VALEUR","SAISIE_PUCE","SAISIE_ITEM_FR_FR","SAISIE_ITEM_EN_US","SAISIE_ITEM_DE_CH","NB_ITEMS","ORDRE_PUCE","VALEUR_MIN","VALEUR_MAX") AS 
select
    bi.ID_BAREME_ITEM,
    lib.LIBELLE as BAREME_LIBELLE,
    bi.VALEUR as SAISIE_VALEUR,
    bi.PUCE as SAISIE_PUCE,
    lib_loc.LIBELLE_FR_FR as SAISIE_ITEM_FR_FR,
    lib_loc.LIBELLE_EN_US as SAISIE_ITEM_EN_US,
    lib_loc.LIBELLE_DE_CH as SAISIE_ITEM_DE_CH,
    bd.NB_ITEMS,
    bi.ORDRE as ORDRE_PUCE,
    bd.VALEUR_MIN,
    bd.VALEUR_MAX
from T_DI_BAREME_ITEM bi
join T_DI_BAREME_DEF bd on bd.ID_BAREME_DEF = bi.ID_BAREME_DEF
left join T_LIBELLE lib on lib.ID = bd.ID_LIBELLE
left join T_LIBELLE_LOCAL lib_loc on lib_loc.ID = bi.ID_LIBELLE_LOCALISE

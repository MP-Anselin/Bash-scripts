##
SYS_ALERTES
=SQL IFNOTEXIST=SYS_ALERTES=
CREATE TABLE SYS_ALERTES ( CODE VARCHAR2(10), LIBELLE VARCHAR2(50), CODEFAM VARCHAR2(10), DESCRIPTION VARCHAR2(4000), NOM_PAGE VARCHAR2(250), REQUETE VARCHAR2(4000), ORDRE NUMBER(*,0), PRIORITE NUMBER(*,0), TYPEALERTE VARCHAR2(1), FORME VARCHAR2(80), CLFROM VARCHAR2(200), INTER VARCHAR2(1) DEFAULT 'N', NOMAPPLIREQ VARCHAR2(30) DEFAULT '', NOM_PAGE1 VARCHAR2(200) DEFAULT '', NOM_PAGE2 VARCHAR2(200) DEFAULT '', GTA VARCHAR2(1), FORMAT VARCHAR2(1), PRIMARY KEY (CODE) )

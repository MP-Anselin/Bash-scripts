WKF_PROCESSUS
=SQL IFNOTEXIST=WKF_PROCESSUS=
CREATE TABLE WKF_PROCESSUS ( PROC_NUM NUMBER(*,3) NOT NULL ENABLE, PROC_LIBELLE VARCHAR2(150), PROC_DUREE_ESTI NUMBER(*,3), PROC_DATECREA VARCHAR2(14), PROC_LOCKED VARCHAR2(5), PROC_STATUT_DATE_DEB_PREV NUMBER(*,3), PURGE_CLASS VARCHAR2(150), PURGE_ACTIVATE VARCHAR2(5), CODE_USER_DBA VARCHAR2(32) NOT NULL ENABLE, PAGESORTIE_ACTIVATE VARCHAR2(5), TRI_TITLE VARCHAR2(20), ORDRE NUMBER(3,0), PRIMARY KEY (PROC_NUM) )

CREATE TABLE "SYS_DEF_REPLIB" ( "ID_REP" VARCHAR2(10) NOT NULL, "LIBELLE" VARCHAR2(50) NOT NULL, "CHEMIN" VARCHAR2(255), "ACCES_USER" VARCHAR2(5) DEFAULT 'false', PRIMARY KEY(ID_REP))

=SQL IFNOTEXIST=SYS_DEF_REPLIB=ACCES_USER=
ALTER TABLE SYS_DEF_REPLIB ADD ACCES_USER  VARCHAR2(5) DEFAULT 'false'
##
SYS_DEF_REPLIB_ACCES_USER
=SQL=
UPDATE
  SYS_DEF_REPLIB
SET
  ACCES_USER='true'
WHERE
  ID_REP IN
  (
    SELECT
      L.ID_REP
    FROM
      SYS_DEF_REPLIB L ,
      SYS_SEC_USR_WDIR W
    WHERE
      L.ID_REP = W.ID_REP
    AND L.CHEMIN LIKE '\%_DSF_\%%' ESCAPE '\'
    AND
      ( SUBSTR(W.login,LENGTH(W.login)     -3,1)='-'
      OR W.LOGIN IN
        ( SELECT LOGIN FROM SYS_USR_CONFIG WHERE REQUETEMATRIC IS NOT NULL )
      )
  )

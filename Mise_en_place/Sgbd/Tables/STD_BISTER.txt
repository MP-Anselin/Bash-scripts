DECLARE 
nbBister NUMBER;
BEGIN
  SELECT count(*)
  INTO nbBister
  FROM STD_BISTER;
 
  IF nbBister = 0 THEN
    INSERT INTO STD_BISTER (CODE, LIBELLE) VALUES ('B', 'BIS');
    INSERT INTO STD_BISTER (CODE, LIBELLE) VALUES ('T', 'TER');
    INSERT INTO STD_BISTER (CODE, LIBELLE) VALUES ('Q', 'QUATER');
    INSERT INTO STD_BISTER (CODE, LIBELLE) VALUES ('C', 'QUINTER');
  END IF;
END;

UPDATE_DATA_NAME_MP_ALP_CODE_SOCIETE
=SQL=
BEGIN
    FOR R IN (
        SELECT TABLE_NAME, COLUMN_NAME 
        FROM USER_TAB_COLS 
        WHERE TABLE_NAME NOT IN (SELECT VIEW_NAME FROM USER_VIEWS) 
        AND COLUMN_NAME IN ('NOM_DONNEE_UX','DATA_NAME_MP_ALP') 
        ORDER BY TABLE_NAME, COLUMN_NAME 
    )
    LOOP
        EXECUTE IMMEDIATE 'UPDATE '||R.TABLE_NAME ||' SET '||R.COLUMN_NAME||' = ''legalEntityID'' WHERE '||R.COLUMN_NAME||' = ''payrollGroupCode''' ;
    END LOOP;
END;

TGE_CREATE_TRIGGER
=SQL=
BEGIN
  FOR TABLE_DIC IN (SELECT TABLE_NAME FROM USER_TABLES WHERE TABLE_NAME LIKE 'DIC%' AND NOT REGEXP_LIKE(TABLE_NAME,'DIC.*_[0-9]+')) LOOP
    TGE_CREATE_TRIGGER(TABLE_DIC.TABLE_NAME);
  END LOOP;
END;

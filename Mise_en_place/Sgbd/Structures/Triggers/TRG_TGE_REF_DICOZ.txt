TRG_TGE_REF_DICOZ
=SQL=
CREATE OR REPLACE TRIGGER TRG_TGE_REF_DICOZ 
BEFORE INSERT OR UPDATE ON TGE_REF_DICOZ

FOR EACH ROW

BEGIN

IF :NEW.GROUPE in ('ATJ','HCO','IND') and :NEW.OCC1='12' THEN

:NEW.OCC1:='13';

END IF;

IF :NEW.GROUPE ='HIS' and :NEW.OCC1='0' THEN

:NEW.OCC1:='18';

END IF;

END TRG_TGE_REF_DICOZ;

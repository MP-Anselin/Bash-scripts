##
EXTR_HRO_CORR_DROP_IDX
=SQL=
CREATE OR REPLACE
PROCEDURE EXTR_HRO_CORR_DROP_IDX
IS
  K INTEGER;
BEGIN
  SELECT COUNT(*)
  INTO K
  FROM user_indexes
  WHERE index_name = 'IDX_EXTR_HRO_CORR';
  IF K             > 0 THEN
    EXECUTE_IMMEDIATE_STRING ('DROP INDEX IDX_EXTR_HRO_CORR ',0);
  END IF;
END ;
